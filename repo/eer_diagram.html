EER Model (Chen Notation Style)

        flowchart TD
            %% Styles
            classDef entity fill:#fff,stroke:#000,stroke-width:2px;
            classDef relation fill:#fff,stroke:#000,stroke-width:2px,shape:diamond;
            classDef attribute fill:#fff,stroke:#000,stroke-width:1px,stroke-dasharray: 5 5;
            classDef attributeKey fill:#fff,stroke:#000,stroke-width:1px,text-decoration:underline;

            %% ================= USERS =================
            USER[USERS]:::entity
            userId([id]):::attributeKey
            userEmail([email]):::attribute
            userRole([role]):::attribute
            userFullname([full_name]):::attribute
            
            USER --- userId
            USER --- userEmail
            USER --- userRole
            USER --- userFullname

            %% ================= BUILDINGS =================
            BUILDING[BUILDINGS]:::entity
            buildId([id]):::attributeKey
            buildLoc([location]):::attribute
            buildName([name]):::attribute
            
            BUILDING --- buildId
            BUILDING --- buildLoc
            BUILDING --- buildName

            %% REL: MANAGES (User - Building)
            MANAGES{MANAGES}:::relation
            USER ---|1,N| MANAGES
            MANAGES ---|1,N| BUILDING

            %% ================= TENANTS =================
            TENANT[TENANTS]:::entity
            tenantId([id]):::attributeKey
            tenantNum([tenant_number]):::attribute
            aptNum([apartment_number]):::attribute
            
            TENANT --- tenantId
            TENANT --- tenantNum
            TENANT --- aptNum

            %% REL: LIVES_IN (User - Tenant - Building)
            %% Modeled as: User is a Tenant in a Building
            IS_TENANT{IS_TENANT}:::relation
            USER ---|1,1| IS_TENANT
            IS_TENANT ---|0,1| TENANT
            
            LOCATED_IN{LOCATED_IN}:::relation
            TENANT ---|1,1| LOCATED_IN
            LOCATED_IN ---|0,N| BUILDING

            %% ================= SERVICERS =================
            SERVICER[SERVICERS]:::entity
            servId([id]):::attributeKey
            servName([full_name]):::attribute
            servProf([profession]):::attribute
            
            SERVICER --- servId
            SERVICER --- servName
            SERVICER --- servProf

            %% ================= MALFUNCTIONS =================
            MALFUNCTION[MALFUNCTIONS]:::entity
            malId([id]):::attributeKey
            malTitle([title]):::attribute
            malStatus([status]):::attribute
            
            MALFUNCTION --- malId
            MALFUNCTION --- malTitle
            MALFUNCTION --- malStatus

            %% REL: REPORTS (Tenant reports Malfunction)
            REPORTS{REPORTS}:::relation
            TENANT ---|1,N| REPORTS
            REPORTS ---|1,1| MALFUNCTION

            %% REL: ASSIGNED_TO (Servicer assigned to Malfunction)
            ASSIGNED{ASSIGNED}:::relation
            SERVICER ---|0,N| ASSIGNED
            ASSIGNED ---|0,N| MALFUNCTION

            %% ================= INTERVENTIONS =================
            INTERVENTION[INTERVENTIONS]:::entity
            intId([id]):::attributeKey
            intStatus([status]):::attribute
            intDate([scheduled_at]):::attribute
            
            INTERVENTION --- intId
            INTERVENTION --- intStatus
            INTERVENTION --- intDate

            %% REL: HAS_INTERVENTION (Malfunction has Intervention)
            HAS_INT{HAS_INT}:::relation
            MALFUNCTION ---|1,N| HAS_INT
            HAS_INT ---|1,1| INTERVENTION

            %% REL: PERFORMS (Servicer performs Intervention)
            PERFORMS{PERFORMS}:::relation
            SERVICER ---|1,N| PERFORMS
            PERFORMS ---|0,N| INTERVENTION

            %% ================= RATINGS =================
            RATING[RATINGS]:::entity
            rateId([id]):::attributeKey
            rateScore([score]):::attribute
            rateComment([comment]):::attribute
            
            RATING --- rateId
            RATING --- rateScore
            RATING --- rateComment

            %% REL: RATED_BY (User rates Intervention)
            GIVES_RATING{GIVES}:::relation
            USER ---|1,N| GIVES_RATING
            GIVES_RATING ---|1,1| RATING

            RATES_INT{RATES}:::relation
            RATING ---|1,1| RATES_INT
            RATES_INT ---|0,N| INTERVENTION
